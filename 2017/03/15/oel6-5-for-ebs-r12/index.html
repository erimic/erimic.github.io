EBS R12 Linux服务器准备

发表于 2017-03-15 | | 阅读次数

参考Metalink文档Note Id 761566.1

准备

假设

  • 数据库APPS用户
  • 目标DB 版本是Oracle Database Enterprise 11.2.0.4
  • 数据库APPS用户

术语

术语 说明
APPSPWD 数据库APPS用户口令
APPSUSER 数据库APPS用户

EBS 12.1.3 CERTIFIED OS

OEL6.5安装介质下载

OEL6.5安装及配置

OEL6.5安装

安装VMTool

mkdir /cdrom
mount /dev/cdrom /cdrom
cd /cdrom/
mkdir /ISO
cp -r * /ISO
cd /ISO
tar -xf VMwareTools-9.6.1-1378637.tar.gz
cd vmware-tools-distrib/
perl vmware-install.pl

参数配置

配置/ETC/SYSCONFIG/NETWORK

vi /etc/sysconfig/network

HOSTNAME=app01

配置/ETC/HOSTS

vi /etc/hosts

127.0.0.1   localhost localhost.localdomain localhost4 localhost4.localdomain4
::1         localhost localhost.localdomain localhost6 localhost6.localdomain6
192.168.228.130 app01.test.com app01

移除/ETC/SYSCONFIG/NETWORKING/PROFILES/DEFAULT/NETWORK

mv /etc/sysconfig/networking/profiles/default/network /etc/sysconfig/networking/profiles/default/network.old

cd 6078836
mv /usr/lib/libdb.so.2 /usr/lib/libdb.so.2.6078836
cp libdb.so.2 /usr/lib

LINK TO MOTIF LIBRARY IN ORACLE APPLICATION SERVER 10.1.2 (ON ORACLE LINUX 5, RHEL 5, ORACLE LINUX 6 AND RHEL 6 ONLY)

unlink /usr/lib/libXtst.so.6
ln -s /usr/X11R6/lib/libXtst.so.6.1 /usr/lib/libXtst.so.6

创建用户及组

#建立组
/usr/sbin/groupadd -g 600 oinstall 
/usr/sbin/groupadd -g 602 dba
/usr/sbin/groupadd -g 603 applgrp
#检查组
cat /etc/group

#建立用户
/usr/sbin/useradd -u 601 -g oinstall -G dba oraprod
/usr/sbin/useradd -u 602 -g applgrp applprod
/usr/sbin/useradd -u 701 -g oinstall -G dba oradev
/usr/sbin/useradd -u 702 -g applgrp appldev
/usr/sbin/useradd -u 801 -g oinstall -G dba oratrn
/usr/sbin/useradd -u 802 -g applgrp appltrn
/usr/sbin/useradd -u 901 -g oinstall -G dba orasit
/usr/sbin/useradd -u 902 -g applgrp applsit
/usr/sbin/useradd -u 911 -g oinstall -G dba orademo
/usr/sbin/useradd -u 912 -g applgrp appldemo
/usr/sbin/useradd -u 921 -g oinstall -G dba orauat
/usr/sbin/useradd -u 922 -g applgrp appluat
#如果用户存在
/usr/sbin/usermod -g oinstall -G dba oracle
#如果nobody用户不存在
/usr/sbin/useradd nobody
cat /etc/passwd
